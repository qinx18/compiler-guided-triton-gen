#define COLS 256

float wall[ROWS][COLS];
float src[COLS];
float dst[COLS];

void pathfinder_kernel() {
  int t, n;
  float min_val;
#pragma scop
  /* Initialize src from first row */
  for (n = 0; n < COLS; n++) {
    src[n] = wall[0][n];
  }

  for (t = 0; t < ROWS - 1; t++) {
    for (n = 0; n < COLS; n++) {
